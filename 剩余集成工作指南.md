#### 步骤 6：修改 fetchList 函数

找到 fetchList 函数（约第 239 行），修改为：
```typescript
async function fetchList(reset: boolean) {
  if (loading.value) return;
  loading.value = true;
  if (reset) page.value = 1;
  
  const params = {
    keyword: keyword.value || undefined,
    ageRange: ageRange.value || undefined,
    seriesId: seriesId.value || undefined,
    sortBy: sortBy.value,
    order: 'desc',
    page: page.value,
    pageSize: 20
  };

  try {
    let data;
    if (activeMediaType.value === 'PICTURE_BOOK') {
      // 调用绘本 API
      data = await getPictureBookList(params);
    } else {
      // 调用音频/视频 API
      params.mediaType = activeMediaType.value;
      data = await getMediaList(params);
    }
    
    if (reset) {
      list.value = data.records;
    } else {
      list.value.push(...data.records);
    }
    finished.value = list.value.length >= data.total;
  } catch (error) {
    showToast('加载失败');
  } finally {
    loading.value = false;
  }
}
```

```typescript
onMounted(async () => {

async function loadFavoriteStatus() {
  try {
    const status = await checkFavoriteStatus(bookId)
    isFavorite.value = status.isFavorited
  } catch (error) {
    // 忽略错误
  }
}

async function onToggleFavorite() {
  try {
    if (isFavorite.value) {
      await removeFavorite(bookId)
      isFavorite.value = false
      showToast('已取消收藏')
    } else {
      await addFavorite(bookId, 'PICTURE_BOOK')
      isFavorite.value = true
      showToast('已收藏')
    }
  } catch (error) {
    showToast('操作失败')
  }
}
```

#### 添加收藏 API

在 `frontend/src/api/favorite.ts` 中添加：
```typescript
import http from './http'

export function addFavorite(resourceId: number, resourceType: string = 'AUDIO') {
  return http.post<void>(`/user/favorites/${resourceId}`, null, {
    params: { resourceType }
  })
}

export function removeFavorite(resourceId: number) {
  return http.delete<void>(`/user/favorites/${resourceId}`)
}

export function checkFavoriteStatus(resourceId: number) {
  return http.get<{ isFavorited: boolean }>(`/user/favorites/${resourceId}/status`)
}
```
